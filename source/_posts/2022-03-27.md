```rust
// src/resources/board_options.rs

use bevy::prelude::Vec3;
use serde::{Deserialize, Serialize};

/// 格子大小设置
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum TileSize {
    /// 固定大小格子
    Fixed(f32),
    /// 窗口响应式格子
    Adaptive { min: f32, max: f32 },
}

/// 计分板位置设置
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum BoardPosition {
    /// 居中
    Centered { offset: Vec3 },
    /// 自定义位置
    Custom(Vec3),
}

/// 计分板生成选项 必须作为resource使用
// 通过序列化保存和加载选项
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BoardOptions {
    /// 地图大小
    pub map_size: (u16, u16),
    /// 地雷数量
    pub bomb_count: u16,
    /// 计分板位置
    pub position: BoardPosition,
    /// 格子大小
    pub tile_size: TileSize,
    /// 格子间距
    pub tile_padding: f32,
    /// 是否启用点开的第一个格子必然是安全格子
    pub safe_start: bool,
}
```

下面为这些结构提供一些默认值
```rust
// src/resources/board_options.rs
```

# 地砖大小

地砖的大小不能是一成不变的, 所以我们需要一个计算地砖大小与地图大小关系的方法. 将如下内容添加到 `BoardPlugin` 的方法中

```rust
	/// 根据地图大小计算格子大小, 自适应属性
    fn adaptative_tile_size(
        window: Res<WindowDescriptor>,
        (min, max): (f32, f32),      // Tile size constraints
        (width, height): (u16, u16), // Tile map dimensions
    ) -> f32 {
        let max_width = window.width / width as f32;
        let max_heigth = window.height / height as f32;
        max_width.min(max_heigth).clamp(min, max)
    }
```

接下来我们在 `create_board` 方法中调用此功能

```rust
// 定义格子大小
        let tile_size = match options.tile_size {
            TileSize::Fixed(v) => v,
            TileSize::Adaptive { min, max } => Self::adaptative_tile_size(
                window,
                (min, max),
                (tile_map.width(), tile_map.height()),
            ),
        };
```

### 面板属性

接下来我们在 `create_board` 方法中添加如下内容

```rust
		// 计算完整面板大小
        let board_size = Vec2::new(
            tile_map.width() as f32 * tile_size,
            tile_map.height() as f32 * tile_size,
        );
        log::info!("board size: {}", board_size);
        // 面板板锚点位置（左下）
        let board_position = match options.position {
            BoardPosition::Centered { offset } => {
                Vec3::new(-(board_size.x / 2.), -(board_size.y / 2.), 0.) + offset
            }
            BoardPosition::Custom(p) => p,
        };
```

>注意：这里选择将面板锚定在左下角而不是中心, 相当于坐标原点位于左下角, 这样所有地砖的坐标相对就都是正数了

```rust
// src/lib.rs

pub mod components;
pub mod resources;

use bevy::log;
use bevy::prelude::*;

use resources::tile_map::TileMap;
use resources::BoardOptions;
use resources::BoardPosition;
use resources::TileSize;

use components::Coordinates;

/// 面板插件
pub struct BoardPlugin;

impl Plugin for BoardPlugin {
    fn build(&self, app: &mut App) {
        app.add_startup_system(Self::create_board);
        log::info!("Loaded Board Plugin");
    }
}

impl BoardPlugin {
    /// 生成面板
    pub fn create_board(
        mut commands: Commands,
        board_options: Option<Res<BoardOptions>>,
        window: Res<WindowDescriptor>,
    ) {
        let mut tile_map = TileMap::empty(20, 20);
        tile_map.set_bombs(40);
        #[cfg(feature = "debug")]
        log::info!("{}", tile_map.console_output());

        // 生成选项是可选的, 因此如果未设置可以选择直接返回默认值
        let options = match board_options {
            None => BoardOptions::default(),
            Some(o) => o.clone(),
        };

        // 地图生成
        let mut tile_map = TileMap::empty(options.map_size.0, options.map_size.1);
        tile_map.set_bombs(options.bomb_count);
        #[cfg(feature = "debug")]
        // 地图调试选项
        log::info!("{}", tile_map.console_output());

        // 定义格子大小
        let tile_size = match options.tile_size {
            TileSize::Fixed(v) => v,
            TileSize::Adaptive { min, max } => Self::adaptative_tile_size(
                window,
                (min, max),
                (tile_map.width(), tile_map.height()),
            ),
        };

        // 计算完整面板大小
        let board_size = Vec2::new(
            tile_map.width() as f32 * tile_size,
            tile_map.height() as f32 * tile_size,
        );
        log::info!("board size: {}", board_size);
        // 面板板锚点位置（左下）
        let board_position = match options.position {
            BoardPosition::Centered { offset } => {
                Vec3::new(-(board_size.x / 2.), -(board_size.y / 2.), 0.) + offset
            }
            BoardPosition::Custom(p) => p,
        };

        // 生成面板
        commands
            .spawn()
            .insert(Name::new("Board"))
            .insert(Transform::from_translation(board_position))
            .insert(GlobalTransform::default())

            .with_children(|parent| {
                // 以计分板中间为锚点生成计分板背景
                parent
                    .spawn_bundle(SpriteBundle {
                        sprite: Sprite {
                            color: Color::WHITE,
                            custom_size: Some(board_size),
                            ..Default::default()
                        },
                        transform: Transform::from_xyz(board_size.x / 2., board_size.y / 2., 0.),
                        ..Default::default()
                    })
                    .insert(Name::new("Background"));
            })

            .with_children(|parent| {
                // 格子
                for (y, line) in tile_map.iter().enumerate() {
                            for (x, tile) in line.iter().enumerate() {
                                parent
                                    .spawn_bundle(SpriteBundle {
                                        sprite: Sprite {
                                            color: Color::GRAY,
                                            custom_size: Some(Vec2::splat(
                                                tile_size - options.tile_padding as f32,
                                            )),
                                            ..Default::default()
                                        },
                                        transform: Transform::from_xyz(
                                            (x as f32 * tile_size) + (tile_size / 2.),
                                            (y as f32 * tile_size) + (tile_size / 2.),
                                            1.,
                                        ),
                                        ..Default::default()
                                    })
                                    .insert(Name::new(format!("Tile ({}, {})", x, y)))
                                    // 添加坐标组件到格子实体
                                    .insert(Coordinates {
                                        x: x as u16,
                                        y: y as u16,
                                    });
                            }
                        }
            });

    }

    /// 根据地图大小计算格子大小, 自适应属性
    fn adaptative_tile_size(
        window: Res<WindowDescriptor>,
        (min, max): (f32, f32),      // Tile size constraints
        (width, height): (u16, u16), // Tile map dimensions
    ) -> f32 {
        let max_width = window.width / width as f32;
        let max_heigth = window.height / height as f32;
        max_width.min(max_heigth).clamp(min, max)
    }
}
```

```rust
// src/lib.rs

/// 周围生成地雷数
fn bomb_count_text_bundle(count: u8, font: Handle<Font>, size: f32) -> Text2dBundle {
    // We retrieve the text and the correct color
    let (text, color) = (
        count.to_string(),
        match count {
            1 => Color::WHITE,
            2 => Color::GREEN,
            3 => Color::YELLOW,
            4 => Color::ORANGE,
            _ => Color::PURPLE,
        },
    );
    // 生成一段文本
    Text2dBundle {
        text: Text {
            sections: vec![TextSection {
                value: text,
                style: TextStyle {
                    color,
                    font,
                    font_size: size,
                },
            }],
            alignment: TextAlignment {
                vertical: VerticalAlign::Center,
                horizontal: HorizontalAlign::Center,
            },
        },
        transform: Transform::from_xyz(0., 0., 1.),
        ..Default::default()
    }
}
```

再为 `BoardPlugin` 创建一个生成生成地砖贴图的方法

```rust
// src/lib.rs

fn spawn_tiles(
    parent: &mut ChildBuilder,
    tile_map: &TileMap,
    size: f32,
    padding: f32,
    color: Color,
    bomb_image: Handle<Image>,
    font: Handle<Font>,
) {
    // 地砖
    for (y, line) in tile_map.iter().enumerate() {
        for (x, tile) in line.iter().enumerate() {
            let coordinates = Coordinates {
              x: x as u16,
              y: y as u16,
            };
            let mut cmd = parent.spawn();
            cmd.insert_bundle(SpriteBundle {
                sprite: Sprite {
                    color,
                    custom_size: Some(Vec2::splat(size - padding)),
                    ..Default::default()
                },
                transform: Transform::from_xyz(
                    (x as f32 * size) + (size / 2.),
                    (y as f32 * size) + (size / 2.),
                    1.,
                ),
                ..Default::default()
            })
            .insert(Name::new(format!("Tile ({}, {})", x, y)))
            .insert(coordinates);
        }
    }
}
```

```rust
use resources::tile::Tile;

// ..
match tile {// 放入spawn_tiles方法内层for循环下部
    // 如果是地雷就添加对应组件和贴图
    Tile::Bomb => {
        cmd.insert(Bomb);
        cmd.with_children(|parent| {
            parent.spawn_bundle(SpriteBundle {
                sprite: Sprite {
                    custom_size: Some(Vec2::splat(size - padding)),
                    ..Default::default()

                transform: Transform::from_xyz(0., 0., 1.),
                texture: bomb_image.clone(),
                ..Default::default()

        });

    // 如果是地雷周围的格子及空格子就添加对应文本的组件
    Tile::BombNeighbor(v) => {
        cmd.insert(BombNeighbor { count: *v });
        cmd.with_children(|parent| {
            parent.spawn_bundle(Self::bomb_count_text_bundle(
                *v,
                font.clone(),

        });
```
